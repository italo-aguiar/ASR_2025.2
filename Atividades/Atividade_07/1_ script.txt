docker network create mysql-network

docker run -d \
  --name mysql-server \
  --network mysql-network \
  -e MYSQL_ROOT_PASSWORD=senha123 \
  -e MYSQL_DATABASE=meu_banco \
  -p 3306:3306 \
  mysql:8.0

docker run -d \
  --name phpmyadmin \
  --network mysql-network \
  -e PMA_HOST=mysql-server \
  -e PMA_PORT=3306 \
  -p 8080:80 \
  phpmyadmin:latest
